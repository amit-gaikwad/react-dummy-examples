create CounterReducer.js in reducers Folder
import {
  INCREMENT_COUNTER,
  DECREMENT_COUNTER,
  SET_COUNTER,
  RESET_COUNTER,
} from "../Actions/ActionTypes";

const initialState = {
  counter: 0,
};

export const counterReducer = (state = initialState, action) => {
  switch (action.type) {
    case INCREMENT_COUNTER:
      return { counter: state.counter + 1 };
    case DECREMENT_COUNTER:
      return { counter: state.counter - 1 };
    case RESET_COUNTER:
      return { counter: initialState.state };
    case SET_COUNTER:
      return { counter: action.value };
    default:
      return state;
  }
};
